\section{Definitions}

Ginzburg-Landau Hamiltonian with core size $\varepsilon$: $E_\varepsilon (u) = \int_\Omega \frac{1}{2} |\nabla u|^2 + \frac{1}{4 \varepsilon^2} (1- |u|^2)^2$ \\
Renormalized energy: $W(a,d) =  - \pi \sum_{1 \leq i \neq j \leq N} d_i d_j \log |a_i - a_j| - \pi \sum_j d_j R(a_j ; a, d)$ \\
Boundary terms in the renormalized energy $R(x; a,d)$ which solves
\begin{align}
  &\Delta R = 0 \text{, in } \Omega, \quad R = - \sum^N_{j=1} d_j \log |x-a_j| \text{, on } \partial \Omega
\end{align} \\
Supercurrents $j(\psi) = \frac{1}{2i}(\overline{\psi} \nabla - \psi \nabla \overline{\psi})$ \\
Vorticity: $J \psi = \frac{1}{2} \nabla \times j(\psi)$ \\

\section{Analytical results of $\varepsilon \to 0$}
In the singular limit $\varepsilon \to 0$, the vorticity concentrates as $J\psieps(t) \;\to\; \pi\sum_{j=1}^N d_j\,\delta_{a_j(t)}$,
and the vortex positions $a(t) = (a_1(t),\ldots,a_N(t))$ solve the
\emph{point vortex system}
  \begin{align}
    \dot{a}_j(t)
  = -\frac{1}{\pi}\,d_j\,\symp\,\nabla_{a_j} W(a(t), d),
  \qquad a_j(0) = a_j^0,
  \end{align}
where $\symp = \begin{pmatrix} 0 & 1 \\ -1 & 0 \end{pmatrix}$.

\section{Vortex tracking}

\subsection{Computing $R$}
The solution $R$ is harmonic 
Harmonic Polynomials are used to solve the PDE. These are given as $h_1 = 1$,
$h_{2m} = \Re (x+ i y)^m$
and $h_{2m + 1} =  \Im (x+ iy)^m$ for $(x,y) \in \mathbb R^2$.

$R$ can be solved numerically like this:
\begin{enumerate}
\item Choose a maximum degree $n$ and fix it.
Compute $(\mathbb{P}_n g)(\exp (i\theta)) = \sum^n_{k=-n} \hat g (k \exp (i k \theta))$
with $\hat g (k) = \frac{1}{2\pi} \int^{2\pi}_0 g(\exp (i\theta)) \exp (- i k \theta) \dd \theta$
\item Compute the Fourier coefficients of the Dirichlet boundary condition up to order $n$ for instance using a FFT
\item Compute the approximate solution $R_n$ as the harmonic expansion of $\mathbb{P}_n g_a$
\begin{align}
    R_n (r \exp (i \theta)) = \sum^n_{k=-n} r^{|k|} \hat g_a (k) \exp (i k \theta)
\end{align}
\end{enumerate}

\subsection{Method}

\textbf{Step 1: Constructing well-prepared initial conditions}
Define initial vortex positions and degrees. Set up the initial phase $H$ such that the Neumann boundary condition is satisfied.
Compute the radial function $f_\varepsilon$.
\\
\textbf{Step 2: evolve vortex positions via Hamiltonian ODE}
solve the ODE using RK4 and compute $R$.
\\
\textbf{Step 3: Smoothing}
At time $t>0$ build back an approximation of the wave function from the vortex positions $a(t)$ as
\begin{align}
    u^* (x; a(t), d) = \exp (i H(x)) \prod^N_{j=1} (\frac{x-a_j}{|x-a_j|})^{d_j}
\end{align}
Smoothing is then applied to get the approximation of the wave function at time $t$ as:
\begin{align}
    \psieps^* (x; a^0, d) = u^* (x; a^0, d) \prod^N_{j=1} f_\varepsilon (|x- a^0_j|), \quad x \in \overline \Omega
\end{align}

Key insight: $\epsilon$ only enters through the 1D preprocessing.
It is no longer a discretization bottleneck.

\begin{lemma}
  Let $\Omega \subset \R^2$ be the unit disk, let $\{a(t)\}_{t\geq 0}$ denote the exact solution to the Hamiltonian mechanics.
  Let $b(t)$ be the approximated ODE trajectory with the same degrees and initial positions $b^0$ with harmonic polynomials of degree up to $n$
  and time step $\delta_t$. Suppose that for some $T>0$,
  \begin{align}
  \rho_T = \min_{t\leq T} \min_{j\neq k} \{ |b_j (t) - b_k  (t)|, \text{dist} (b_k(t) , \partial \Omega), |a_j (t) - a_k  (t)|, \text{dist} (a_j(t) , \partial \Omega)\}>0
  \end{align}
  is such that $0 < \rho_T < 1$. Then we have:
  \begin{align}
    |a(t) - b(t)| \leq C (|a^0 -b^0| + (1- \rho_T)^n n^{1-l}) + C \delta t^4
  \end{align}
\end{lemma}
